sql комманда CREATE:

CREATE TABLE используется для создания новой таблицы с заданным именем
в текущей базе данных


CREATE DATABASE используется для создания базы данных с указанным именем




sql комманда SELECT:

SELECT FROM используется для того что бы извлекать данные из таблицы


sql комманда UPDATE:


UPDATE используется для изменения  имеющихся данных в таблице


sql комманда DROP:

DROP TABLE  используется для удаления
обьектов в базе данных

для удаления таблицы используется DROP TABLE
для удаления индекса используется DROP INDEX
для удаления баззы данных используется DROP DATABASE
